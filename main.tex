\begin{nav_map}{deg=54, min=0}{deg=54, min=20}{deg=7, min=40}{deg=8, min=0}{0.1}
    \fill (10, 4) circle (0.5);
    \fill (4, 10) circle (0.25);

    % Calculate relative coordinates
    \navConvertCoords{1}{deg=54, min=1, sec=36}
    \navConvertCoords{0}{deg=7, min=40, sec=12}
    \fill (\navGetAxis{x}, \navGetAxis{y}) circle (.1);


    % Calculate relative coordinates
    \navConvertCoords{1}{deg=54, min=3}
    \navConvertCoords{0}{deg=7, min=58}

    % Pillar buoys
    \navMarks{x=\navGetAxis{x} - 20, y=\navGetAxis{y} + 10, scaling=1}{}
    \navMarks{x=\navGetAxis{x} - 10, y=\navGetAxis{y} + 10, scaling=1}{buoyArea=fill, top=circle}
    \navMarks{x=\navGetAxis{x}, y=\navGetAxis{y} + 10, scaling=1}{buoyArea=fill, top=circle, topArea=fill}

    % Conical
    \navMarks{x=\navGetAxis{x} - 20, y=\navGetAxis{y}, scaling=1}{buoy=conical}
    \navMarks{x=\navGetAxis{x} - 10, y=\navGetAxis{y}, scaling=1}{buoy=conical, buoyArea=fill, top=up}
    \navMarks{x=\navGetAxis{x}, y=\navGetAxis{y}, scaling=1}{buoy=conical, buoyArea=fill, top=up, topArea=fill}

    % Cylindrical
    \navMarks{x=\navGetAxis{x} - 20, y=\navGetAxis{y} - 10, scaling=1}{buoy=cylindrical}
    \navMarks{x=\navGetAxis{x} - 10, y=\navGetAxis{y} - 10, scaling=1}{buoy=cylindrical, buoyArea=fill, top=box}
    \navMarks{x=\navGetAxis{x}, y=\navGetAxis{y} - 10, scaling=1}{buoy=cylindrical, buoyArea=fill, top=box, topArea=fill}

    % Spherical buoy
    \navMarks{x=\navGetAxis{x} - 20, y=\navGetAxis{y} + 20, scaling=1}{buoy=spherical, lightSide=left, lightColor=green}

    % Spar buoy
    \navMarks{x=\navGetAxis{x} - 10, y=\navGetAxis{y} + 20, scaling=1}{buoy=sparbuoy, buoyArea=fill, top=box, lightSide=right, lightColor=red}

    % Barrel buoy
    \navMarks{x=\navGetAxis{x} - 0, y=\navGetAxis{y} + 20, scaling=1}{buoy=barrelbuoy, lightSide=right}

    % Super buoy
    \navMarks{x=\navGetAxis{x} - 20, y=\navGetAxis{y} + 30, scaling=1}{buoy=superbuoy}

    % Mooring buoy
    \navMarks{x=\navGetAxis{x} - 10, y=\navGetAxis{y} + 30, scaling=1}{buoy=mooringbuoy}

    % LightHouse
    \navMarks{x=\navGetAxis{x} - 0, y=\navGetAxis{y} + 30, scaling=1}{buoy=lighthouse, topArea=fill}

    \navLightHouseFire{color=green, start angle=35, end angle=93}

    \navLightHouseFire{color=white, start angle=93, end angle=225}

    \navLightHouseFire{color=red, start angle=225, end angle=330}

    % cardinalmarks
    \navMarks{x=25, y=25, scaling=1}{top=north, topArea=fill}
    \navMarks{x=35, y=15, scaling=1}{top=east, topArea=fill}
    \navMarks{x=25, y=5, scaling=1}{top=south, topArea=fill}
    \navMarks{x=15, y=15, scaling=1}{top=west, topArea=fill}


    % Custom position symbol
    \navMarkPosition{x=\navGetAxis{x} - 50, y=\navGetAxis{y} + 15, scaling=1}
\end{nav_map}
