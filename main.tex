\begin{nav_map}{deg=54, min=0}{deg=54, min=20}{deg=7, min=40}{deg=8, min=0}{0.1}
    \fill (10, 4) circle (0.5);
    \fill (4, 10) circle (0.25);

    % Calculate relative coordinates
    \navConvertCoords{1}{deg=54, min=1, sec=36}
    \navConvertCoords{0}{deg=7, min=40, sec=12}
    \fill (\navGetCoord{x}, \navGetCoord{y}) circle (.1);


    % Calculate relative coordinates
    \navConvertCoords{1}{deg=54, min=3}
    \navConvertCoords{0}{deg=7, min=58}

    % Pillar buoys
    \navMarks{x=\navGetCoord{x} - 20, y=\navGetCoord{y} + 10, scaling=1}{}
    \navMarks{x=\navGetCoord{x} - 10, y=\navGetCoord{y} + 10, scaling=1}{buoyArea=fill, top=circle}
    \navMarks{x=\navGetCoord{x}, y=\navGetCoord{y} + 10, scaling=1}{buoyArea=fill, top=circle, topArea=fill}

    % Conical
    \navMarks{x=\navGetCoord{x} - 20, y=\navGetCoord{y}, scaling=1}{buoy=conical}
    \navMarks{x=\navGetCoord{x} - 10, y=\navGetCoord{y}, scaling=1}{buoy=conical, buoyArea=fill, top=up}
    \navMarks{x=\navGetCoord{x}, y=\navGetCoord{y}, scaling=1}{buoy=conical, buoyArea=fill, top=up, topArea=fill}

    % Cylindrical
    \navMarks{x=\navGetCoord{x} - 20, y=\navGetCoord{y} - 10, scaling=1}{buoy=cylindrical}
    \navMarks{x=\navGetCoord{x} - 10, y=\navGetCoord{y} - 10, scaling=1}{buoy=cylindrical, buoyArea=fill, top=box}
    \navMarks{x=\navGetCoord{x}, y=\navGetCoord{y} - 10, scaling=1}{buoy=cylindrical, buoyArea=fill, top=box, topArea=fill}

    % Spherical buoy
    \navMarks{x=\navGetCoord{x} - 20, y=\navGetCoord{y} + 20, scaling=1}{buoy=spherical}

    % Spar buoy
    \navMarks{x=\navGetCoord{x} - 10, y=\navGetCoord{y} + 20, scaling=1}{buoy=sparbuoy, buoyArea=fill, top=box}

    % Barrel buoy
    \navMarks{x=\navGetCoord{x} - 0, y=\navGetCoord{y} + 20, scaling=1}{buoy=barrelbuoy}

    % Super buoy
    \navMarks{x=\navGetCoord{x} - 20, y=\navGetCoord{y} + 30, scaling=1}{buoy=superbuoy}

    % Mooring buoy
    \navMarks{x=\navGetCoord{x} - 10, y=\navGetCoord{y} + 30, scaling=1}{buoy=mooringbuoy}

    % cardinalmarks
    \navMarks{x=25, y=25, scaling=1}{top=north}
    \navMarks{x=35, y=15, scaling=1}{top=east}
    \navMarks{x=25, y=5, scaling=1}{top=south}
    \navMarks{x=15, y=15, scaling=1}{top=west}
\end{nav_map}
